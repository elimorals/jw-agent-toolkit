=== README.md ===
# my-agent

An agent plugin for [`jw-agent-toolkit`](https://github.com/eliascipre/jw-agent-toolkit) (Fase 41 plugin SDK).

## Install (editable)

```bash
uv sync
uv run pytest
```

## Verify discovery

```bash
uv pip install -e .
uv run jw plugins list
uv run jw plugins verify my_agent
```

## Implement your logic

Edit `src/my_agent/agent.py`. The stub returns a placeholder finding;
replace with real calls to `jw-core` / `jw-rag` / `jw-agents`.

See [cookbook recipes](https://github.com/eliascipre/jw-agent-toolkit/tree/main/docs/cookbook)
for copy-pasteable examples.

## Scope reminder

`jw-agent-toolkit` is 100% for Jehovah's Witnesses publications. Build plugins
that serve the JW community: meeting prep, apologetics, ministry tools,
specialized parsers for JW formats, etc.

"""my-agent — stub agent. Replace with real logic.

See https://github.com/eliascipre/jw-agent-toolkit/tree/main/docs/cookbook
for recipes showing how to call jw-core / jw-rag / jw-agents APIs.
"""

from __future__ import annotations

from typing import Any


async def my_agent(
    *,
    question: str = "",
    language: str = "en",
    **kwargs: Any,
) -> dict[str, Any]:
    """Stub agent. Returns an AgentResult-shaped dict.

    The Protocol requires:
      - `__name__: str` (callables get this for free).
      - async `__call__(**kwargs) -> AgentResult-like`.

    Optional attributes can be attached after definition (cf. capabilities matrix
    in docs/plugin-sdk/capabilities.md).
    """

    return {
        "findings": [
            {
                "source": "stub",
                "text": f"TODO: implement my_agent logic for {question!r}",
                "citation": {
                    "url": "https://wol.jw.org/",
                    "title": "Placeholder citation",
                    "metadata": {},
                },
            },
        ],
        "metadata": {"agent": "my_agent", "language": language},
    }


# Optional plugin attributes (detected via hasattr at use-site).
my_agent.languages = ["en", "es", "pt"]  # type: ignore[attr-defined]
my_agent.version = "0.1.0"  # type: ignore[attr-defined]

"""Smoke + contract tests for my-agent.

Three deterministic tests, no network. CI-green from the first commit.
"""

from __future__ import annotations

from my_agent.agent import my_agent


async def test_smoke() -> None:
    result = await my_agent(question="Trinity", language="en")
    assert result["findings"], "agent must return at least one finding"


async def test_contract_shape() -> None:
    result = await my_agent(question="x", language="en")
    for finding in result["findings"]:
        assert finding["source"]
        assert finding["text"]
        assert finding["citation"]
        assert finding["citation"]["url"].startswith("https://")


async def test_citations_present() -> None:
    result = await my_agent(question="x", language="en")
    for finding in result["findings"]:
        assert finding["citation"]


async def test_optional_attributes_exposed() -> None:
    # Capability matrix: optional `languages` and `version` (since 1.0).
    assert hasattr(my_agent, "languages")
    assert hasattr(my_agent, "version")
